from sklearn.metrics import precision_score, recall_score, confusion_matrix
import pandas as pd

# Function to compute Specificity
def specificity_score(y_true, y_pred):
    cm = confusion_matrix(y_true, y_pred)
    tn, fp, fn, tp = cm.ravel()
    return tn / (tn + fp)

# Initialize the best models
best_models = {
    "KNN": best_knn_model,
    "Logistic Regression": best_logic_model,
    "Decision Tree": best_tree_model
}

# Compute metrics for each model on the test set
results = []
for model_name, model in best_models.items():
    # Make predictions
    y_pred = model.predict(X_test)

    # Compute metrics
    precision = precision_score(y_test, y_pred)
    recall = recall_score(y_test, y_pred)
    specificity = specificity_score(y_test, y_pred)

    # Append results
    results.append({
        "Model": model_name,
        "Precision": precision,
        "Recall (Sensitivity)": recall,
        "Specificity (True Negative Rate)": specificity
    })

# Convert results to DataFrame for easier visualization

Cohenâ€™s Kappa is a metric that measures the level of agreement between the predicted and actual classes while accounting for the possibility of agreement occurring by chance. Unlike accuracy, which does not consider the role of random chance, Cohenâ€™s Kappa provides a more robust evaluation of classification models, especially for imbalanced datasets.

from sklearn.metrics import cohen_kappa_score
